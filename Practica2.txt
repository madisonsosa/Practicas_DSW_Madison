fun main() {
    //areglo de 5 notas
    val notas = arrayOf(10,9,8,8,9)
    
    //arreglo null
    val notasNull: Array<Int?> = arrayOfNulls(5)
    notasNull[0] = 10
    notasNull[1] = 9
    notasNull[2] = 8
    notasNull[3] = 8
    notasNull[4] = 9
    
    // usa Ã­ndices de notas. pero muestra los valores de notasNull
    for(i in notas.indices){
        println("Indice: " + i + " Valor= " + notasNull[i])
    }
    
    //Calcular promedio
    var suma = 0
    var a = 0
    for(n in notasNull){
        if(n != null){
            suma = suma + n
            a = a + 1
        }
    }
    
    var prom = suma / a
    println("promedio= " + prom)
}
