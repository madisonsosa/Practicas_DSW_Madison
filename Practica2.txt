//Codigo que permita guardar en una lista numeros pares entre 1 y 100
fun main(){
    val numPar = mutableListOf<Int>()

for (i in 1..100) //toma los valores desde 1 hasta 100, uno por uno.
    if (i % 2 == 0) { //revisa si el número es par
        numPar.add(i) //si el número es par lo agrega a la lista
    }
    println(numPar)//imprime la lista
}
